@Service
class PlaceQueryService(
    private val projectRepository: ProjectRepository,
    private val placeRepository: PlaceRepository,
    private val mapper: PlaceDtoMapper,
) {
    @Transactional(readOnly = true)
    fun list(
        projectCode: String,
        pageable: Pageable,
        unitIds: List<String>? = null,
    ): PageResponse<PlaceSummaryDto> {
        val project = projectRepository.findBySlug(projectCode)
            .orElseThrow { NoSuchElementException("Project not found: $projectCode") }

        val page = if (unitIds.isNullOrEmpty()) {
            placeRepository.findByProjectId(project.id, pageable)
        } else {
            val uuidList = unitIds.mapNotNull { parseUUID(it) }
            placeRepository.findByProjectIdAndUnitIds(project.id, uuidList, pageable)
        }

        val dtos = mapper.toSummaryList(page.content)
        return PageResponse.of(page, dtos)
    }
}
